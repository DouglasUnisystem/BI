SELECT (VC.DESCRICAO_PRODUTO || ' - ' || VC.DESC_ARMAZEM) AS PRODUTO_ARMAZEM,
       SUM(VC.MEDIA_COMPRA) AS MEDIA_COMPRA,
       SUM(VC.MEDIA_VENDA) AS MEDIA_VENDA,
       SUM(VC.TOT_QTDE_COMPRA) AS TOT_QTDE_COMPRA,
       SUM(VC.TOT_QTDE_VENDA) AS TOT_QTDE_VENDA,
       SUM(VC.SALDO_DISPONIVEL) AS SALDO_DISPONIVEL,
       VC.FILIAL,
       VC.NOME_EMPRESA,
       VC.APELIDO_SAFRA,
       VC.DESC_ARMAZEM,
       VC.DESCRICAO_PRODUTO
  FROM VW_POS_EST_COMERCIALIZACAO VC
 WHERE 1 = 1
   &EMPRESA
   &SAFRA
   &FILIAL
   &ARMAZEM
   &PRODUTO
   &DATA_INI
   &DATA_FIM
 GROUP BY
       VC.FILIAL,
       VC.NOME_EMPRESA,
       VC.APELIDO_SAFRA,
       VC.DESC_ARMAZEM,
       VC.DESCRICAO_PRODUTO
