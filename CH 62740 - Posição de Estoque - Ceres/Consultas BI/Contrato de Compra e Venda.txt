SELECT 'COMPRA' AS TIPO,
       CC.ID,
       CC.PRODUTOR_COMPRADOR,
       CC.DATA,
       CC.NR_CONTRATO,
       CC.QUANTIDADE,
       CC.UNITARIO,
       CC.VALOR_TOTAL       
  FROM VW_CONTRATO_COMPRA CC
 WHERE 1 = 1    
   &EMPRESA
   &SAFRA
   &FILIAL
   &PRODUTO
   &DATA_INI
   &DATA_FIM   
UNION
SELECT 'VENDA' AS TIPO,
       CV.ID,
       CV.CLIENTE_ENTREGA AS PRODUTOR_COMPRADOR,       
       CV.DATA,
       CV.NR_CONTRATO,
       CV.QUANTIDADE,
       CV.UNITARIO,
       CV.VALOR_TOTAL
  FROM VW_CONTRATO_VENDA  CV
 WHERE 1 = 1    
   &EMPRESA
   &SAFRA
   &FILIAL
   &PRODUTO
   &DATA_INI
   &DATA_FIM
