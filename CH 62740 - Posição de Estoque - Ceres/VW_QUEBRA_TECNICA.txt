CREATE MATERIALIZED VIEW VW_QUEBRA_TECNICA REFRESH FORCE ON DEMAND START WITH SYSDATE NEXT SYSDATE + 5/24/60 AS
SELECT QT.SEQ_PLA_CLIENTE AS SEQ_PLA_ENDERECO,
       SUM(DECODE(QT.QUANTIDADE,NULL,0,QT.QUANTIDADE)) AS QUEBRA_TECNICA,
       QT.COD_EMPRESA,
       QT.COD_FILIAL,
       QT.COD_SAFRA,
       QT.SEQ_PLA_ARMAZEM,
       QT.SEQ_PLA_PRODUTO,
       QT.DATA
  FROM AGRICOLA.QUEBRA_TECNICA QT,
       AGRICOLA.PRODUTOS       PD,
       AGRICOLA.SUB_GRUPO      SG
 WHERE 1 = 1
   AND QT.SEQ_PLA_PRODUTO   = PD.SEQ_PLA_PRODUTO  (+)
   AND PD.SEQ_PLA_SUB_GRUPO = SG.SEQ_PLA_SUB_GRUPO(+)
 GROUP BY QT.SEQ_PLA_CLIENTE,
       QT.COD_EMPRESA,
       QT.COD_FILIAL,
       QT.COD_SAFRA,
       QT.SEQ_PLA_ARMAZEM,
       QT.SEQ_PLA_PRODUTO,
       QT.DATA
