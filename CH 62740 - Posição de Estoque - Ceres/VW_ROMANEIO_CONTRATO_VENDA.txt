CREATE MATERIALIZED VIEW VW_ROMANEIO_CONTRATO_VENDA REFRESH FORCE ON DEMAND START WITH SYSDATE NEXT SYSDATE + 5/24/60 AS
SELECT LNR.SEQ_PLA_NOTA,
       SUM(NVL(EEI.QUANTIDADE, 0)) AS QUANTIDADE,
       SUM(NVL(EER.PESO_BRUTO, 0)) - SUM(NVL(EER.PESO_TARA, 0)) AS QTDE_TOTAL
  FROM AGRICOLA.EST_LIGA_NOTA_ROMANEIO LNR,
       AGRICOLA.EST_ENTRADAS_ROM       EER,
       AGRICOLA.EST_ENTRADAS           EEE,
       AGRICOLA.EST_ENTRADAS_ITENS     EEI
 WHERE EER.SEQ_PLA_ENTRADA = LNR.SEQ_PLA_ROMANEIO
   AND EER.SEQ_PLA_ENTRADA = EEE.SEQ_PLA_ENTRADA
   AND EEE.SEQ_PLA_ENTRADA = EEI.SEQ_PLA_ENTRADA
 GROUP BY LNR.SEQ_PLA_NOTA
