CREATE MATERIALIZED VIEW VW_TRANSF_ENTRADAS REFRESH FORCE ON DEMAND START WITH SYSDATE NEXT SYSDATE + 5/24/60 AS
SELECT TR.SEQ_PLA_BENEFICIADO AS SEQ_PLA_ENDERECO,
       SUM(DECODE(TR.QUANTIDADE,NULL,0,TR.QUANTIDADE)) AS TRANSFERENCIA,
       TR.EMPRESA_DESTINO AS COD_EMPRESA,
       TR.FILIAL_DESTINO AS COD_FILIAL,
       TR.COD_SAFRA,
       TR.SEQ_PLA_ARMAZEM,
       TR.SEQ_PLA_PRODUTO,
       TR.DATA_LANCAMENTO AS DATA
  FROM AGRICOLA.TRANSFERENCIAS TR,
       AGRICOLA.PRODUTOS       PD,
       AGRICOLA.SUB_GRUPO      SG
 WHERE 1 = 1
   AND TR.SEQ_PLA_PRODUTO         = PD.SEQ_PLA_PRODUTO   (+)
   AND PD.SEQ_PLA_SUB_GRUPO       = SG.SEQ_PLA_SUB_GRUPO (+)
 GROUP BY TR.SEQ_PLA_BENEFICIADO,TR.EMPRESA_DESTINO,
       TR.FILIAL_DESTINO,
       TR.COD_SAFRA,
       TR.SEQ_PLA_ARMAZEM,
       TR.SEQ_PLA_PRODUTO,
       TR.DATA_LANCAMENTO
