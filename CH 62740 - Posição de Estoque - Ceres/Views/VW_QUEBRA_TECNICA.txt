CREATE OR REPLACE VIEW VW_QUEBRA_TECNICA AS
SELECT QT.SEQ_PLA_CLIENTE AS SEQ_PLA_ENDERECO,
       SUM(DECODE(QT.QUANTIDADE,NULL,0,QT.QUANTIDADE)) AS QUEBRA_TECNICA,
       QT.COD_EMPRESA,
       QT.COD_FILIAL,
       QT.COD_SAFRA,
       QT.SEQ_PLA_ARMAZEM,
       QT.SEQ_PLA_PRODUTO,
       QT.DATA
  FROM CERES2311.QUEBRA_TECNICA QT,
       CERES2311.PRODUTOS       PD,
       CERES2311.SUB_GRUPO      SG
 WHERE 1 = 1
   AND QT.SEQ_PLA_PRODUTO   = PD.SEQ_PLA_PRODUTO  (+)
   AND PD.SEQ_PLA_SUB_GRUPO = SG.SEQ_PLA_SUB_GRUPO(+)
 GROUP BY QT.SEQ_PLA_CLIENTE,
       QT.COD_EMPRESA,
       QT.COD_FILIAL,
       QT.COD_SAFRA,
       QT.SEQ_PLA_ARMAZEM,
       QT.SEQ_PLA_PRODUTO,
       QT.DATA;
