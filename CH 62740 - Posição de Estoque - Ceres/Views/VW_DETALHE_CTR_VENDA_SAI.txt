CREATE OR REPLACE VIEW VW_DETALHE_CTR_VENDA_SAI AS
SELECT FIL.SIGLA_FILIAL,
       TRUNC(ES.DATA_EMISSAO) DATA_EMISSAO,
       ES.NR_DOCUMENTO,
       SUM(NVL(ESI.QUANTIDADE,0)) QUANTIDADE,
       CASE WHEN SUM(NVL(ESI.QUANTIDADE, 0)) > 0 THEN (AVG(ES.VALOR_TOTAL) / SUM(NVL(ESI.QUANTIDADE, 0))) ELSE 0 END UNITARIO,
       AVG(ES.VALOR_TOTAL) VALOR_TOTAL,
       (AVG(ES.VALOR_TOTAL)-SUM(NVL(ESV.VALOR_FUNRURAL, 0)+NVL(ESV.VALOR_SENAR, 0)+NVL(ESV.VALOR_FETHAB, 0)+NVL(ESV.VALOR_IMAFIR, 0)+NVL(ESV.VALOR_FACS, 0)+NVL(ESV.VALOR_RAT, 0) + NVL(ESV.VALOR_IAGRO,0))) VALOR_LIQUIDO,
       SUM(ESV.VALOR_FUNRURAL) VALOR_FUNRURAL,
       SUM(ESV.VALOR_SENAR) VALOR_SENAR,
       SUM(ESV.VALOR_RAT) VALOR_RAT,
       SUM(ESV.VALOR_FETHAB) VALOR_FETHAB,
       SUM(ESV.VALOR_FACS) VALOR_FACS,
       SUM(ESV.VALOR_IAGRO) VALOR_IAGRO,
       SUM(ESV.VALOR_ICMS) VALOR_ICMS,
       ES.COD_TIPO_LCTO,
       (SELECT R.NR_ROMANEIO
          FROM CERES2311.EST_SAIDAS_ROM R
         WHERE R.SEQ_PLA_SAIDA = ES.SEQ_PLA_SAIDA) AS NR_ROMANEIO,
       NVL(ER.DEVOLVIDO,'N') AS DEVOLVIDO,
       '' AS NR_DOCUMENTO_DEVOLUCAO,
       SUM(NVL(ESI.QUANTIDADE,0)) AS QTD_NOTA,
       SUM(NVL(ER.PESO_BRUTO-ER.PESO_TARA,0)) AS QTDE_TOTAL,
       NVL(ROUND(LF.QUANTIDADE*(LF.VALOR_TONELADA/1000),2),0) AS VALOR_FRETE,
       SUM(ESV.VALOR_IMAFIR) AS VALOR_IMAFIR,
       SUM((NVL(ER.PESO_BRUTO,0)-NVL(ER.PESO_TARA,0)) - NVL(ER.DES_UMIDADE,0) - NVL(ER.DES_IMPUREZA,0) - NVL(ER.DES_AVARIADO,0) - NVL(ER.DES_ARDIDO,0) - NVL(ER.DES_OUTROS,0)) AS QTDE_LIQUIDA,
       SUM(DECODE(CD.PESO_ORIGEM_DESTINO,'D',DECODE(EC.SEQ_PLA_SAI_ITEM,NULL,0,NVL(EC.PESO_CHEGADA,0)),0)) AS PESO_CHEGADA,
       CD.SEQ_PLA_PED_VENDA AS ID
  FROM CERES2311.CONTRATO_DESMEMBRAMENTO CD,
       CERES2311.EST_SAIDAS              ES,
       CERES2311.EST_SAIDAS_ITENS        ESI,
       CERES2311.EST_SAIDAS_VALORES      ESV,
       CERES2311.FILIAIS                 FIL,
       CERES2311.CLIENTES_ENDERECOS      CE,
       CERES2311.EST_SAIDAS_ROM          ER,
       CERES2311.EST_LIGA_FRETE_ROMANEIO LF,
       CERES2311.EST_PESO_CHEGADA        EC
 WHERE CD.SEQ_PLA_CONT_DES     = ES.SEQ_PLA_CONT_DES
   AND ES.SEQ_PLA_SAIDA        = ESI.SEQ_PLA_SAIDA
   AND ESI.SEQ_PLA_SAI_ITEM    = ESV.SEQ_PLA_SAI_ITEM
   AND ES.COD_FILIAL           = FIL.COD_FILIAL
   AND ES.COD_EMPRESA          = FIL.COD_EMPRESA
   AND ES.SEQ_PLA_ENDERECO     = CE.SEQ_PLA_ENDERECO
   AND ES.SEQ_PLA_SAIDA        = ER.SEQ_PLA_SAIDA    (+)
   AND ES.SEQ_PLA_SAIDA        = LF.SEQ_PLA_SAIDA    (+)
   AND ESI.SEQ_PLA_SAI_ITEM    = EC.SEQ_PLA_SAI_ITEM (+)
   AND NVL(ES.CANCELADO,'N') = 'N'
 GROUP BY FIL.SIGLA_FILIAL,
          TRUNC(ES.DATA_EMISSAO),
          ES.NR_DOCUMENTO,
          ES.COD_TIPO_LCTO,
          ES.SEQ_PLA_SAIDA,
          NVL(ER.DEVOLVIDO,'N'),
          ROUND(LF.QUANTIDADE*(LF.VALOR_TONELADA/1000),2),
          CD.SEQ_PLA_PED_VENDA;
