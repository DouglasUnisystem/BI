CREATE OR REPLACE VIEW VW_RETORNO_CONTRATO_VENDA AS
SELECT LR.SEQ_PLA_ENT_ITEM,
       ES.NR_DOCUMENTO,
       NVL(ESR.PESO_BRUTO,0) - NVL(ESR.PESO_TARA,0) AS SUBTOTAL
  FROM CERES2311.EST_SAIDAS       ES,
       CERES2311.EST_SAIDAS_ITENS ESI,
       CERES2311.LIGA_NF_RETORNO  LR,
       CERES2311.EST_SAIDAS_ROM   ESR
 WHERE ES.SEQ_PLA_SAIDA    = ESI.SEQ_PLA_SAIDA
   AND LR.SEQ_PLA_SAI_ITEM = ESI.SEQ_PLA_SAI_ITEM
   AND ES.SEQ_PLA_SAIDA    = ESR.SEQ_PLA_SAIDA (+)
