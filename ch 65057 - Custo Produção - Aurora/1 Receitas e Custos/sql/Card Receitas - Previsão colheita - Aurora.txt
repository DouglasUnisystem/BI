SELECT MAX((SELECT E.NOME_EMPRESA FROM AGRICOLA.EMPRESAS E WHERE TO_CHAR(E.COD_EMPRESA) LIKE TO_CHAR(DECODE( NVL(:COD_EMPRESA_STRING,'0') ,'0','%', :COD_EMPRESA_STRING )) AND ROWNUM = 1)) AS Empresa,
       ROUND(SUM(SC_HA  * QTD_HA),2) * 60 AS TOTAL --SACAS DE 60 FIXO       
  FROM VW_DADOS_COLHEITA DC
 WHERE 1 = 1
   AND TO_CHAR(DC.COD_SAFRA)               LIKE TO_CHAR(DECODE( NVL(:COD_SAFRA_STRING  ,'0') ,'0','%', :COD_SAFRA_STRING   )) --PARAM
   &CULTURA
