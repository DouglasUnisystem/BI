SELECT DATA_LCTO AS DESCRICAO,
       DATA_LCTO AS ID,
       DATA_LCTO AS VALOR
  FROM (SELECT TRUNC(CP.DATA_LCTO) AS DATA_LCTO
          FROM GUSTAVO.CONTAS_PAGAR CP
         UNION
        SELECT TRUNC(CR.DATA_LCTO) AS DATA_LCTO
          FROM GUSTAVO.CONTAS_RECEBER CR  
       ) GG  
GROUP BY DATA_LCTO
ORDER BY DATA_LCTO
